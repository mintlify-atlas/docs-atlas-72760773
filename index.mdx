## Quick Example

```typescript
import { GrvtClient, EGrvtEnvironment, ECurrency } from '@grvt/sdk';

// Initialize the client
const client = new GrvtClient({
  apiKey: 'your-api-key',
  apiSecret: 'your-api-secret',
  env: EGrvtEnvironment.DEV,
});

// Get account summary
const summary = await client.getFundingAccountSummary();

// Transfer funds
const transfer = await client.transfer({
  from_account_id: 'from-id',
  to_account_id: 'to-id',
  currency: ECurrency.USDT,
  num_tokens: '100',
});
```
